192
193 listening = true;
194 try {

202 while(!finalized) {
203 MPI_Comm client;
204 if (MPI_Comm_accept(portname, MPI_INFO_NULL, 0, MPI_COMM_SELF, &client) != MPI_SUCCESS) {
205 MTCL_MPIP2P_ERROR("listen_threadF: MPI_Comm_accept error\n");

210 MPI_Comm_remote_size(client, &remote_size);
211 MTCL_MPIP2P_PRINT(100, "remote size is %d\n", remote_size);

213 /* Retrieve the remote size and create one handle for each of the
214 * connecting rank */
215
216 REMOVE_CODE_IF(std::unique_lock ulock(shm));
217 comm_sizes.insert({client, remote_size});
218 REMOVE_CODE_IF(ulock.unlock());

222 REMOVE_CODE_IF(ulock.lock());
223 connections.insert({handle, false});
224 REMOVE_CODE_IF(ulock.unlock());
225 addinQ(true, handle);

227 }
228 }
