32 ssize_t probe(size_t& size, const bool blocking=true) {
33 if (probed.first){
34 size = probed.second;
35 return (size ? (ssize_t)sizeof(size_t) : 0);
36 }
37 ssize_t sz;
38 if (blocking) {
39 sz = in.getsize();
40 } else {
41 sz = in.trygetsize();
42 if (sz<0 && errno == EAGAIN) {
43 errno = EWOULDBLOCK;
44 return -1;
45 }
46 }
47 if (sz<0) return -1;
48 size = (size_t)sz;
49 probed = {true, size};
50
51 // If EOS, consume it immediately so that subsequent probes do not see it.
52 if (size == 0) {
53 char dummy;
54 (void)in.tryget(&dummy, 1);
55 }
56 return (size ? (ssize_t)sizeof(size_t) : 0);
57 }
58
59 ssize_t receive(void* buff, size_t size) {
60 size_t probedSize;
61 if (!probed.first){
62 ssize_t r = probe(probedSize, true);
63 if (r <= 0) return r;
64 } else {
65 probedSize = probed.second;
66 }
67 if (probedSize == 0) {
68 probed = {false, 0};
69 return 0;
70 }
71 if (probedSize > size){
72 errno = ENOMEM;
73 return -1;
74 }
75 probed = {false, 0};
76 return in.get(buff, probedSize);
77 }

79 bool peek() {
80 ssize_t r = in.peek();
81 return (r > 0);
82 }
