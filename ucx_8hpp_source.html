548  void update() {
549  REMOVE_CODE_IF(std::unique_lock ulock(shm, std::defer_lock));
550 
551  REMOVE_CODE_IF(ulock.lock());
552  tmpset = set;
553  REMOVE_CODE_IF(ulock.unlock());
554 
555  struct timeval wait_time = {.tv_sec = 0, .tv_usec=UCX_POLL_TIMEOUT};
556  int nready = 0;
557 
558  // Only if we are listening for new connections
559  if(fdmax != -1) {
560  switch (nready=select(fdmax+1, &tmpset, NULL, NULL, &wait_time)) {
561  case -1: {
562  if(errno == EBADF) {
563  MTCL_UCX_PRINT(100, "ConnUCX::update select ERROR: errno=EBADF\n");
564  return;
565  }
566  MTCL_UCX_ERROR("ConnUCX::update select ERROR: errno=%d -- %s\n", errno, strerror(errno));
567  }
568  case 0:
569  break;
570  }
571 
572  // Only for new connections on the OOB socket
573  if(FD_ISSET(this->listen_sck, &tmpset)) {
574  int connfd = accept(this->listen_sck, (struct sockaddr*)NULL ,NULL);
575  if (connfd == -1){
576  MTCL_UCX_ERROR("ConnUCX::update accept ERROR: errno=%d -- %s\n", errno, strerror(errno));
577  return;
578  }

580  ucp_address_t* peer_addr;
581  size_t peer_addr_len;
582  int res = exchange_address(connfd, local_addr, local_addr_len, &peer_addr, &peer_addr_len);
583  if(res != 0) {
584  MTCL_UCX_PRINT(100, "ConnUCX::update error exchanging address\n");
585  return;
586  }
587 
588  // We only use connfd for the OOB handshake
589  close(connfd);
590 
591  ucs_status_t ep_status = UCS_OK;
592  ucp_ep_h client_ep;
593  ucp_ep_params_t ep_params;
594  ep_params.field_mask = UCP_EP_PARAM_FIELD_REMOTE_ADDRESS |
595  UCP_EP_PARAM_FIELD_USER_DATA;
596  ep_params.address = peer_addr;
597  ep_params.user_data = &ep_status;
598 
599  ep_status = ucp_ep_create(ucp_worker, &ep_params, &client_ep);
600  if(ep_status != UCS_OK) {
601  MTCL_UCX_PRINT(100, "ConnUCX::update error creating the endpoint\n");
602  return;
603  }
604  free(peer_addr);
605 
606  HandleUCX* handle = new HandleUCX(this, client_ep, ucp_worker);
607 
608  REMOVE_CODE_IF(ulock.lock());
609  connections.insert({client_ep, {handle, false}});
610  addinQ(true, handle);
611  REMOVE_CODE_IF(ulock.unlock());
612  }
613  }
614 
615  // Poll on managed endpoints to detect something is ready to be read
616  // This will cause, at the same time, progress on the user handles
617  size_t size = -1;
618  size_t max_eps = connections.size();
619 
620  // One-shot progress
621  int prog = -1;
622  prog = ucp_worker_progress(ucp_worker);
623  (void)prog;
624 
625  // The polling mechanism provided by ucp_stream_worker_poll does not
626  // return twice the same endpoint if no NEW network events are detected.
627  // This means, with repeated polling if some network event is detected
628  // but not managed by the user (by a receive/probe) the endpoint will
629  // not be displayed again even if data are ready to be read
630  ucp_stream_poll_ep_t* ready_eps = new ucp_stream_poll_ep_t[max_eps];
631  size = ucp_stream_worker_poll(ucp_worker, ready_eps, max_eps, 0);
632  if(size < 0) {
633  MTCL_UCX_PRINT(100, "ConnUCX::update error in ucp_stream_worker_poll\n");
634  return;
635  }
636 
637  REMOVE_CODE_IF(ulock.lock());
638  // Some of the endpoints are ready, we need to check if we are managing that ep
639  for(size_t i=0; i<size; i++) {
640  ucp_stream_poll_ep_t ep = ready_eps[i];
641 
642  auto it = connections.find(ep.ep);
643  if(it != connections.end()) {
644  auto& handlePair = it->second;
645  if(handlePair.second) {
646  handlePair.second = false;
647  addinQ(false, handlePair.first);
648  }
649  }
650  }
651  REMOVE_CODE_IF(ulock.unlock());
652 
653  delete[] ready_eps;
654  return;
655  }
