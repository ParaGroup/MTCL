73static inline void mtcl_cpu_relax(void) {
74#if defined(__x86_64__) || defined(__i386__)
75#define MTCL_PAUSE() __asm__ __volatile__("pause")
76#elif defined(__aarch64__) || defined(__arm__)
77 #define MTCL_PAUSE() __asm__ __volatile__("yield")
78#elif defined(__powerpc__) || defined(__ppc__) || defined(__PPC__)
79 #define MTCL_PAUSE() __asm__ __volatile__("or 27,27,27")
80#elif defined(__riscv)
81// compile with -DMTCL_RISCV_HAS_PAUSE=1 to switch to "pause" if supported
82 #if defined(MTCL_RISCV_HAS_PAUSE)
83 #define MTCL_PAUSE() __asm__ __volatile__("pause")
84 #else
85 #define MTCL_PAUSE() __asm__ __volatile__("nop")
86 #endif
87#else
88// portable fallback
89 #define MTCL_PAUSE() __asm__ __volatile__("")
90#endif
91
92 MTCL_PAUSE();
93}

121template<typename H>
122static inline ssize_t nb_probe_with_timeout(H* h, size_t& size, int retry, unsigned timeout_ms) {
123 static_assert(std::is_pointer_v<decltype(h)>, "h must be a pointer");
124 // This check ensures H has a method probe(size_t&, bool) returning ssize_t
125 static_assert(std::is_same_v<decltype(h->probe(size, false)), ssize_t>,
126 "H must implement: ssize_t probe(size_t& size, bool blocking)");
127 if (retry <= 0) retry = 1;
128
129 using clock = std::chrono::steady_clock;
130
131 const auto start = clock::now();
132 const auto deadline = start + std::chrono::milliseconds((uint64_t)timeout_ms * (uint64_t)retry);
133 constexpr auto SPIN_BUDGET = std::chrono::microseconds(SPIN_THRESHOLD);
134 constexpr auto MAX_SLEEP = std::chrono::microseconds(WAIT_INTERNAL_TIMEOUT);
135
136 auto sleep = std::chrono::microseconds(1);
137
138 while (true) {
139 const ssize_t r = h->probe(size, false);
140
141 if (r > 0) return r;
142 if (r == 0) { errno = ECONNRESET; return -1; }
143
144 if (errno != EWOULDBLOCK && errno != EAGAIN) return -1;
145
146 const auto now = clock::now();
147 if (now >= deadline) { errno = ETIMEDOUT; return -1; }
148
149 const auto elapsed = now - start;
150
151 if (elapsed < SPIN_BUDGET) { // fast path
152 mtcl_cpu_relax();
153 } else {
154 // Backoff to reduce CPU usage on slow peers or half-open connections
155 std::this_thread::sleep_for(sleep);
156 sleep = std::min(sleep * 2, MAX_SLEEP);
157 }
158 }
159}

164static inline int internal_connect(int sockfd, const struct sockaddr *addr, socklen_t addrlen) {
165 int saved_errno=0;
166 int rc = 0;
167 // Set O_NONBLOCK
168 int sockfd_flags_before;
169 if((sockfd_flags_before=fcntl(sockfd,F_GETFL,0)<0)) return -1;
170 if(fcntl(sockfd,F_SETFL,sockfd_flags_before | O_NONBLOCK)<0) return -1;
171 // Start connecting (asynchronously)
172 do {
173 if (connect(sockfd, addr, addrlen)<0) {
174 // Did connect return an error? If so, we'll fail.
175 if ((errno != EWOULDBLOCK) && (errno != EINPROGRESS)) {
176 rc = -1;
177 saved_errno=errno;
178 } else { // Otherwise, we'll wait for it to complete.
179 saved_errno= EHOSTUNREACH;
180
181 // Set a deadline timestamp 'timeout' ms from now (needed b/c poll can be interrupted)
182 struct timespec now;
183 if(clock_gettime(CLOCK_MONOTONIC, &now)<0) { rc=-1; break; }
184 struct timespec deadline = { .tv_sec = now.tv_sec,
185 .tv_nsec = now.tv_nsec + (UNREACHABLE_ADDR_TIMOUT*1000000l)};
186 // Wait for the connection to complete.
187 do {
188 // Calculate how long until the deadline
189 if(clock_gettime(CLOCK_MONOTONIC, &now)<0) {rc=-1; break;}
190
191 int ms_until_deadline = (int)( (deadline.tv_sec - now.tv_sec)*1000l
192 + (deadline.tv_nsec - now.tv_nsec)/1000000l);
193 if(ms_until_deadline<0) { rc=0; break; }
194 // Wait for connect to complete (or for the timeout deadline)
195 struct pollfd pfds[] = { { .fd = sockfd, .events = POLLOUT } };
196
197 rc = poll(pfds, 1, ms_until_deadline);
198 // If poll 'succeeded', make sure it *really* succeeded
199 if(rc>0) {
200 int error = 0; socklen_t len = sizeof(error);
201 int retval = getsockopt(sockfd, SOL_SOCKET, SO_ERROR, &error, &len);
202 if(retval==0) saved_errno = error;
203 if(error!=0) rc=-1;
204 }
205 } while(rc==-1 && errno==EINTR); // If poll was interrupted, try again.
206 if(rc==0) rc=-1; // Did poll timeout? If so, fail.
207 }
208 }
209 } while(0);
210 // Restore original O_NONBLOCK state
211 if(fcntl(sockfd,F_SETFL,sockfd_flags_before)<0) return -1;
212 errno = saved_errno;
213 return rc;
214}

217static inline int internal_connect(const std::string& address, int retry, unsigned timeout_ms) {
218 const std::string host = address.substr(0, address.find(":"));
219 const std::string svc = address.substr(host.length()+1);
220
221 MTCL_PRINT(100, "[MTCL]:", "connecting to %s:%s\n", host.c_str(), svc.c_str());
222
223 int fd=-1;
224 struct addrinfo hints;
225 struct addrinfo *result, *rp;
226
227 memset(&hints, 0, sizeof(hints));
228 hints.ai_family = AF_UNSPEC; /* Allow IPv4 or IPv6 */
229 hints.ai_socktype = SOCK_STREAM; /* Stream socket */
230 hints.ai_flags = 0;
231 hints.ai_protocol = IPPROTO_TCP; /* Allow only TCP */
232
233 // resolve the address (assumo stringa formattata come host:port)
234 if (getaddrinfo(host.c_str(), svc.c_str(), &hints, &result) != 0) {
235 MTCL_PRINT(100, "MTCL:", "internal_connect getaddrinfo error, errno=%d\n", errno);
236 return -1;
237 }
238
239 bool connected=false;
240 do {
241 // try to connect to a possible one of the resolution results
242 for (rp = result; rp != NULL; rp = rp->ai_next) {
243 fd = socket(rp->ai_family, rp->ai_socktype,
244 rp->ai_protocol);
245 if (fd == -1) {
246 MTCL_PRINT(100, "[MTCL]:", "internal_connect socket error, errno=%d\n", errno);
247 continue;
248 }
249
250 if (internal_connect(fd, rp->ai_addr, rp->ai_addrlen) != -1) {
251 retry=0;
252 connected=true;
253 break; /* Success */
254 }
255 close(fd);
256 }
257 if (!connected && retry>0) {
258 --retry;
259 std::this_thread::sleep_for(std::chrono::milliseconds(timeout_ms));
260 MTCL_PRINT(100, "[MTCL]:", "retry to connect to %s:%s\n", host.c_str(), svc.c_str());
261 }
262 } while(retry>0);
263
264 free(result);
265
266 if (rp == NULL) /* No address succeeded */
267 return -1;
268 return fd;
269}
