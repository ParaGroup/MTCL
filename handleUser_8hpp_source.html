99 ssize_t probe(size_t& size, const bool blocking=true) {
100 newConnection = false;
101 if (!isReadable){
102 MTCL_PRINT(100, "[MTCL]:", "HandleUser::probe handle not readable\n");
103 return 0;
104 }
105 if (!realHandle) {
106 MTCL_PRINT(100, "[MTCL]:", "HandleUser::probe EBADF\n");
107 errno = EBADF; // the handle is not valid or closed
108 return -1;
109 }
110 if (realHandle->closed_rd) return 0;
111
112 // reading the header to get the size of the message
113 ssize_t r;
114 if ((r=realHandle->probe(size, blocking))<=0) {
115 switch(r) {
116 case 0: {
117 isReadable=false;
118 realHandle->close(true, true);
119 return 0;
120 }
121 case -1: {
122 if (errno==EINVAL) {
123 return -1;
124 }
125 if (errno==ECONNRESET) {
126 realHandle->close(true, true);
127 return 0;
128 }
129 if (errno==EWOULDBLOCK || errno==EAGAIN) {
130 errno = EWOULDBLOCK;
131 return -1;
132 }
133 }}
134 return r;
135 }
136 if (size==0) { // EOS received
137 realHandle->close(false, true);
138 isReadable=false;
139 return 0;
140 }
141 return r;
142 }
