84  static inline int connectionHandshake(char *& teamID, Handle *h) {
85  // new connection, read handle type (p2p=0, collective=1)
86  size_t size;
87  if (h->probe(size, true) <=0) {
88  MTCL_ERROR("[Manager]:\t", "addinQ handshake error in probe, errno=%d\n", errno);
89  teamID=nullptr;
90  return -1;
91  }
92  int collective = 0;
93  if (h->receive(&collective, sizeof(int)) <=0) {
94  MTCL_ERROR("[Manager]:\t", "addinQ handshake error in receiving collective flag, errno=%d\n", errno);
95  teamID=nullptr;
96  return -1;
97  }
98 
99  // If collective, the handle sends further data with string teamID.
100  // The teamID uniquely associate a single context to all handles of the same collective
101  // This is useful to synchronize the root thread for the accepts.
102  if(collective) {
103  size_t size;
104  if (h->probe(size, true) <= 0) {
105  MTCL_ERROR("[Manager]:\t", "addinQ handshake error in probe, teamID size, errno=%d\n", errno);
106  teamID=nullptr;
107  return -1;
108  }
109  // sanity check
110  if (size>1048576) {
111  MTCL_ERROR("[Manager]:\t", "addinQ handshake error in probe, teamID size TOO LARGE (size=%ld)\n", size);
112  teamID=nullptr;
113  return -1;
114  }
115 
116  teamID = new char[size+1];
117  assert(teamID);
118  if (h->receive(teamID, size) <=0) {
119  MTCL_ERROR("[Manager]:\t", "addinQ handshake error in probe, receiving teamID, errno=%d\n", errno);
120  delete [] teamID;
121  teamID=nullptr;
122  return -1;
123  }
124  teamID[size] = '\0';
125  MTCL_PRINT(100, "[Manager]: \t", "Manager::addinQ received connection for team: %s\n", teamID);
126  }
127  return 0;
