38// token non-blocking completion check with compatibility across Paho versions.
39template <typename Tok>
40auto token_try_wait(const Tok& tok, int) -> decltype(tok->try_wait(), bool()) {
41 return tok->try_wait();
42}
43template <typename Tok>
44bool token_try_wait(const Tok& tok, long) {
45 return tok->wait_for(std::chrono::milliseconds(0));
46}
47template <typename Tok>
48bool token_try_wait(const Tok& tok) {
49 return token_try_wait(tok, 0);
50}

67 if (!tok) { errno = ECOMM; result = 0; return -1; }

75 if (!tok) { errno = ECOMM; return -1; }

78 return 0;
79 }

81 int make_progress() override { int r; return test(r); }

146 };

152 ConnRequestVectorMQTTRecv(size_t sizeHint = 1) : slots(sizeHint) {}

161 for (size_t i = 0; i < currSize; ++i) {
162 if (!slots[i].active) continue;

165 slots[i].active = false;

167 }
168 if (!done) return false;
169 slots[i].active = false;
170 }
171 return true;
172 }

175 for (size_t i = 0; i < currSize; ++i) {
176 if (slots[i].active) {
177 (void)slots[i].req.wait();
178 slots[i].active = false;
179 }
180 }
181 }

201 inline void clearProbed() { probed = {false, 0}; }
202
203 // Drain/discard exactly one *payload* message of size 'payload_sz' from the internal queue.
204 // Precondition: probe() has already consumed the framed MQTT packet and has queued the payload
205 // as {payload_bytes, payload_sz} in 'messages'.
206 bool drainCurrentMessage(size_t payload_sz) {
207 while (!messages.empty()) {
208 auto msg = messages.front();
209 messages.pop();
210
211 // Ignore late out-of-band ACKs
212 if ((size_t)msg.second < HDR_SZ && msg.first == "ack") {
213 continue;
214 }
215 if ((size_t)msg.second != payload_sz) {
216 MTCL_MQTT_PRINT(100, "HandleMQTT::drainCurrentMessage EPROTO, size mismatch (expected %ld got %ld)\n",
217 (long)payload_sz, (long)msg.second);
218 errno = EPROTO;
219 return false;
220 }
221 return true; // drained
222 }
223 MTCL_MQTT_PRINT(100, "HandleMQTT::drainCurrentMessage EPROTO, missing payload after probe\n");
224 errno = EPROTO;
225 return false;
226 }

294 ssize_t receive(void* buff, size_t size){
295 size_t probedSize;
296 if (!probed.first){
297 ssize_t r = probe(probedSize);
298 if (r <= 0) return r;
299 } else {
300 probedSize = probed.second;
301 }
302
303 if (probedSize == 0) {
304 probed={false, 0};
305 return 0;
306 }
307 if (probedSize > size){
308 // Enforce "drain semantics": discard the whole message to allow subsequent receives.
309 MTCL_MQTT_PRINT(100, "HandleMQTT::receive EMSGSIZE, buffer too small -> draining message\n");
310 if (!drainCurrentMessage(probedSize)) {
311 // errno set by drainCurrentMessage()
312 probed = {false, 0};
313 return -1;
314 }
315 probed = {false, 0};
316 errno = EMSGSIZE;
317 return -1;
318 }
319
320 while(true) {
321 if(!messages.empty()) {
322 auto msg = messages.front();
323 messages.pop();
324
325 // Ignore out-of-band connection ACKs that might arrive late.
326 // ACK payloads are plain strings and are not framed (no HDR_SZ).
327 if ((size_t)msg.second < HDR_SZ && msg.first == "ack") {
328 continue;
329 }
330
331 if (msg.second != (ssize_t)probedSize) {
332 MTCL_MQTT_PRINT(100, "HandleMQTT::receive EPROTO, receiving less data or connection reset\n");
333 errno=EPROTO;
334 return -1;
335 }
336 probed={false, 0};
337 memcpy(buff, msg.first.data(), probedSize);
338 return probedSize;
339 }
340
341 mqtt::const_message_ptr msg;
342 bool res = client->try_consume_message(&msg);
343 // A message has been received
344 if(res) {
345 if (!msg) {
346 MTCL_MQTT_PRINT(100, "client reset connection\n");
347 errno = ECONNRESET;
348 return -1;
349 }
350
351 // This topic should have been subscribed upon creation of the
352 // handle object
353 if(msg->get_topic() == in_topic) {
354 const auto& payload = msg->get_payload();
355
356 // Late ACKs are plain strings and must not be delivered to the user.
357 if (payload.size() < HDR_SZ && payload == "ack") {
358 continue;
359 }
360
361 if (payload.size() == probedSize) {
362 probed={false, 0};
363 memcpy(buff, payload.data(), probedSize);
364 return (ssize_t)probedSize;
365 }
366 if (payload.size() < HDR_SZ) {
367 MTCL_MQTT_PRINT(100, "HandleMQTT::receive EPROTO, short packet\n");
368 errno = EPROTO;
369 return -1;
370 }
371 uint64_t szbe = 0;
372 memcpy(&szbe, payload.data(), HDR_SZ);
373 const size_t sz = (size_t)be64toh(szbe);
374 const size_t plsz = payload.size() - HDR_SZ;
375 if (sz != probedSize || plsz != probedSize) {
376 MTCL_MQTT_PRINT(100, "HandleMQTT::receive EPROTO, size mismatch\n");
377 errno = EPROTO;
378 return -1;
379 }
380 probed={false, 0};
381 memcpy(buff, payload.data() + HDR_SZ, probedSize);
382 return (ssize_t)probedSize;
383 }
384 }
385 if constexpr (MQTT_POLL_TIMEOUT)
386 std::this_thread::sleep_for(std::chrono::milliseconds(MQTT_POLL_TIMEOUT));
387 }
388 return 0;
389 }

400 ssize_t probe(size_t& size, const bool blocking=true) {
401 if (probed.first) {
402 size = probed.second;
403 return (size ? (ssize_t)HDR_SZ: 0);
404 }
405 while(!messages.empty()) {
406 // It means the Manager put the message header in the queue for us
407 auto msg = messages.front();
408 messages.pop();
409
410 // Ignore out-of-band connection ACKs that might arrive late
411 if ((size_t)msg.second < HDR_SZ && msg.first == "ack") {
412 continue;
413 }
414
415 if (msg.second < (ssize_t)HDR_SZ) {
416 MTCL_MQTT_PRINT(100, "HandleMQTT::probe EPROTO, receiving less data or connection reset (messages)\n");
417
418 probed = {true, 0};
419 errno = EPROTO;
420 return -1;
421 }
422 uint64_t szbe = 0;
423 memcpy(&szbe, msg.first.data(), HDR_SZ);
424 size = (size_t)be64toh(szbe);
425
426 const size_t totsize = (size_t)msg.second;
427 const size_t plsz = totsize - HDR_SZ;
428 if (plsz != size) {
429 MTCL_MQTT_PRINT(100, "HandleMQTT::probe EPROTO, size mismatch\n");
430 errno = EPROTO;
431 return -1;
432 }
433 if (plsz > 0) {
434 messages.push({msg.first.substr(HDR_SZ, plsz), (ssize_t)plsz});
435 }
436 probed = {true, size};
437 return (size ? (ssize_t)HDR_SZ: 0);
438 }
439
440 mqtt::const_message_ptr msg;
441 bool disconnected = false;
442 auto try_get_header_msg = [&]() -> bool {
443 // try to consume one message
444 if (!client->try_consume_message(&msg)) {
445 return false;
446 }
447 if (!msg) {
448 disconnected = true;
449 return false;
450 }
451
452 // Ignore out-of-band connection ACKs that might arrive late
453 if (msg->get_topic() == in_topic) {
454 const auto& payload = msg->get_payload();
455 if (payload.size() < HDR_SZ && payload == "ack") {
456 return false;
457 }
458 }
459 // accept only the expected topic
460 return (msg->get_topic() == in_topic);
461 };
462
463 if (!blocking) {
464 if (!try_get_header_msg()) {
465 if (disconnected || !client->is_connected()) {
466 MTCL_MQTT_PRINT(100, "client reset connection\n");
467 errno = ECONNRESET;
468 return -1;
469 }
470 errno = EWOULDBLOCK;
471 return -1;
472 }
473 } else { // blocking
474 while(!try_get_header_msg()) {
475 if (disconnected || !client->is_connected()) {
476 MTCL_MQTT_PRINT(100, "client reset connection\n");
477 errno = ECONNRESET;
478 return -1;
479 }
480 if constexpr (MQTT_POLL_TIMEOUT)
481 std::this_thread::sleep_for(std::chrono::milliseconds(MQTT_POLL_TIMEOUT));
482 }
483 }
484 const auto& payload = msg->get_payload();
485 if (payload.size() < HDR_SZ) {
486 MTCL_MQTT_PRINT(100, "HandleMQTT::probe EPROTO, receiving less data (%ld)\n", payload.size());
487 errno = EPROTO;
488 return -1;
489 }
490 uint64_t szbe = 0;
491 memcpy(&szbe, payload.data(), HDR_SZ);
492 size = (size_t)be64toh(szbe);
493
494 const size_t plsz = payload.size() - HDR_SZ;
495 if (plsz != size) {
496 MTCL_MQTT_PRINT(100, "HandleMQTT::probe EPROTO, size mismatch\n");
497 errno = EPROTO;
498 return -1;
499 }
500 if (plsz > 0) {
501 messages.push({payload.substr(HDR_SZ, plsz), (ssize_t)plsz});
502 }
503 probed = {true, size};
504 return (size ? (ssize_t)HDR_SZ: 0);
505 }

525inline int requestMQTTRecv::test(int& result) {
526 if (completed) { result = 1; return 0; }
527
528 size_t sz = 0;
529 const ssize_t pr = h->probe(sz, false);
530 if (pr < 0) {
531 if (errno == EWOULDBLOCK) { result = 0; return 0; }
532 result = 0;
533 return -1;
534 }
535 if (pr == 0) {
536 // EOS
537 completed = true;
538 got = 0;
539 h->clearProbed();
540 result = 1;
541 return 0;
542 }
543
544 if (sz > expected) {
545 // Drain semantics: discard the message and return EMSGSIZE
546 if (!h->drainCurrentMessage(sz)) {
547 // errno set by drainCurrentMessage()
548 h->clearProbed();
549 completed = true;
550 got = (ssize_t)sz;
551 result = 1;
552 return -1;
553 }
554 h->clearProbed();
555 completed = true;
556 got = (ssize_t)sz; // record the real message size
557 errno = EMSGSIZE;
558 result = 1; // completed (with error)
559 return -1;
560 }
561 if (sz == 0) {
562 completed = true;
563 got = 0;
564 h->clearProbed();
565 result = 1;
566 return 0;
567 }
568
569 // probe() already queued the payload in h->messages
570 if (h->messages.empty()) { result = 0; return 0; }
571
572 auto msg = h->messages.front();
573 h->messages.pop();
574
575 if ((size_t)msg.second != sz) {
576 errno = EPROTO;
577 result = 0;
578 return -1;
579 }
580
581 memcpy(buff, msg.first.data(), sz);
582 h->clearProbed();
583 completed = true;
584 got = (ssize_t)sz;
585 result = 1;
586 return 0;
587}

589inline int requestMQTTRecv::wait() {
590 if (completed) return 0;
591
592 // Use probe() in blocking mode to get the real size
593 size_t sz = 0;
594 const ssize_t pr = h->probe(sz, true);
595 if (pr < 0) { got = -1; return -1; }
596 if (pr == 0) { // EOS
597 h->clearProbed();
598 completed = true;
599 got = 0;
600 return 0;
601 }
602
603 if (sz > expected) {
604 if (!h->drainCurrentMessage(sz)) {
605 h->clearProbed();
606 completed = true;
607 got = (ssize_t)sz;
608 return -1; // errno already set (EPROTO)
609 }
610 h->clearProbed();
611 completed = true;
612 got = (ssize_t)sz;
613 errno = EMSGSIZE;
614 return -1;
615 }
616
617 // Now safe: receive() will consume the queued payload and clear probed
618 const ssize_t rr = h->receive(buff, expected);
619 if (rr < 0) { got = -1; return -1; }
620 got = rr;
621 completed = true;
622 return 0;
623}

728 void update() {
729 REMOVE_CODE_IF(std::unique_lock ulock(shm, std::defer_lock));
730
731 // Consume new-connection messages
732 mqtt::const_message_ptr msg;
733 if (listening && newconn_consuming) {
734 newConnClient->try_consume_message(&msg);
735 }
736
737 if (msg) {
738 // New message on new connection topic
739 if(msg->get_topic() == new_connection_topic) {
740 const std::string base = msg->to_string();
741 const mqtt::string topic_in = base + MQTT_IN_SUFFIX;
742 const mqtt::string topic_out = base + MQTT_OUT_SUFFIX;
743 // If the client retries with the same topic, avoid creating duplicates.
744 bool already = false;
745 REMOVE_CODE_IF(ulock.lock());
746 for (auto &kv : connections) {
747 auto* h = kv.first;
748 if (h && h->in_topic == topic_in && h->out_topic == topic_out) {
749 already = true;
750 break;
751 }
752 }
753 REMOVE_CODE_IF(ulock.unlock());
754
755 if (already) {
756 try {
757 auto ack = mqtt::make_message(topic_out, "ack");
758 ack->set_qos(1);
759 newConnClient->publish(ack)->wait();
760 } catch (...) {}
761 } else {
762 mqtt::async_client* aux_cli = new mqtt::async_client(server_address, topic_in);
763 if (!createClient(topic_in, aux_cli)) {
764 delete aux_cli;
765 return;
766 }
767
768 // Must send the ACK to the remote end otherwise we can miss
769 // messages published immediately after the connection request.
770 auto ack = mqtt::make_message(topic_out, "ack");
771 ack->set_qos(1);
772 aux_cli->publish(ack)->wait();
773
774 HandleMQTT* handle = new HandleMQTT(this, aux_cli, topic_out, topic_in);
775
776 REMOVE_CODE_IF(ulock.lock());
777 connections.insert({handle, false});
778 REMOVE_CODE_IF(ulock.unlock());
779 // Never call Manager::addinQ() while holding the protocol lock (shm).
780 addinQ(true, handle);
781 }
782 }
783 }
784 else if (listening && newconn_consuming) {
785 if (!newConnClient->is_connected()) {
786 MTCL_MQTT_PRINT(100, "update: lost connection, waiting a while for reconnecting\n");
787 std::this_thread::sleep_for(std::chrono::milliseconds(MQTT_CONNECT_TIMEOUT));
788 if (!newConnClient->is_connected()) {
789 MTCL_MQTT_PRINT(100, "ConnMQTT::update: no connection yet, keep going...\n");
790 return;
791 }
792 MTCL_MQTT_PRINT(100, "ConnMQTT::update: re-established connection\n");
793 }
794 }
795
796 std::vector<HandleMQTT*> ready;
797 REMOVE_CODE_IF(ulock.lock());
798 for (auto &[handle, to_manage] : connections) {
799 if(to_manage) {
800 mqtt::const_message_ptr msg;
801 const bool res = handle->client->try_consume_message(&msg);
802 if (!res) continue;
803 if (!msg) {
804 handle->messages.push({std::string(), 0});
805 } else {
806 handle->messages.push({msg->get_payload(), (ssize_t)msg->get_payload().length()});
807 }
808 to_manage = false;
809 ready.push_back(handle);
810 }
811 }
812 REMOVE_CODE_IF(ulock.unlock());
813 for(auto* h: ready) addinQ(false, h);
814 }

817 // String for connection composed of manager_id:topic
818 Handle* connect(const std::string& address, int retry, unsigned timeout_ms) {
819 std::string manager_id = address.substr(0, address.find(":"));
820
821 if (manager_id.empty()) {
822 MTCL_MQTT_PRINT(100, "ConnMQTT::connect: server connect string must be defined\n");
823 errno = EINVAL;
824 return nullptr;
825 }
826
827 // Defaults if caller does not specify them
828 const int ntries = (retry < 0) ? CCONNECTION_RETRY : (retry == 0 ? 1 : retry);
829 const unsigned step_ms = (timeout_ms == 0) ? CCONNECTION_TIMEOUT : timeout_ms;
830
831 // Keep the same topic/client-id across retries.
832 const std::string topic = appName + std::to_string(count++);
833 const mqtt::string topic_out = topic + MQTT_OUT_SUFFIX;
834 const mqtt::string topic_in = topic + MQTT_IN_SUFFIX;
835
836 mqtt::connect_options connOpts;
837 connOpts.set_keep_alive_interval(20);
838 connOpts.set_clean_session(true);
839 connOpts.set_user_name(MQTT_USERNAME);
840 connOpts.set_password(MQTT_PASSWORD);
841
842 mqtt::async_client* client = new mqtt::async_client(server_address, topic);
843 try {
844 client->connect(connOpts)->wait();
845 client->subscribe({topic_out}, {0})->wait();
846 client->start_consuming();
847 } catch (...) {
848 errno = ECOMM;
849 try {
850 client->stop_consuming();
851 if (client->is_connected()) client->disconnect()->wait();
852 } catch (...) {}
853 delete client;
854 return nullptr;
855 }
856 auto pubmsg = mqtt::make_message(manager_id + MQTT_CONNECTION_TOPIC, topic);
857 pubmsg->set_qos(1);
858
859 // During connection establishment we must consume only the control ACK.
860 // Any other message arriving on the same subscription (for example due to
861 // QoS mixing or broker reordering) must be preserved and delivered later
862 // through the HandleMQTT message queue, otherwise the stream gets desynchronized.
863 std::vector<std::pair<std::string, ssize_t>> early_msgs;
864 mqtt::const_message_ptr ack;
865 bool got_ack = false;
866 for (int attempt = 0; attempt < ntries && !got_ack; ++attempt) {
867 try {
868 client->publish(pubmsg)->wait();
869 } catch (...) {
870 errno = ECOMM;
871 break;
872 }
873 const auto deadline = std::chrono::steady_clock::now() + std::chrono::milliseconds(step_ms);
874 const auto spin_until = std::chrono::steady_clock::now() + std::chrono::microseconds(200);
875 auto sleep = std::chrono::microseconds(1);
876 const auto max_sleep = std::chrono::microseconds(200);
877
878 while (std::chrono::steady_clock::now() < deadline) {
879 mqtt::const_message_ptr m;
880 const bool res = client->try_consume_message(&m);
881 if (res && m) {
882 const auto& payload = m->get_payload();
883 if (payload.size() < HandleMQTT::HDR_SZ && payload == "ack") {
884 ack = m;
885 got_ack = true;
886 break;
887 }
888 // Not an ACK....
889 early_msgs.emplace_back(payload, (ssize_t)payload.size());
890 }
891
892 if (std::chrono::steady_clock::now() < spin_until) {
893 mtcl_cpu_relax();
894 } else {
895 std::this_thread::sleep_for(sleep);
896 sleep = sleep + sleep;
897 if (sleep > max_sleep) sleep = max_sleep;
898 }
899 }
900
901 if (!got_ack && attempt + 1 < ntries) {
902 std::this_thread::sleep_for(std::chrono::milliseconds(step_ms));
903 }
904 }
905
906 if (!got_ack) {
907 if (errno == 0) errno = ETIMEDOUT;
908 try {
909 client->stop_consuming();
910 if (client->is_connected()) client->disconnect()->wait();
911 } catch (...) {}
912 delete client;
913 return nullptr;
914 }
915
916 // We write on topic + MQTT_IN_SUFFIX and we read from topic + MQTT_OUT_SUFFIX
917 HandleMQTT* handle = new HandleMQTT(this, client, topic_in, topic_out, true);
918 // Preserve any early data/control messages observed during connect.
919 // They will be parsed by probe()/receive() as usual.
920 for (auto& p : early_msgs) {
921 handle->messages.push(std::move(p));
922 }
923 {
924 REMOVE_CODE_IF(std::unique_lock lock(shm));
925 connections[handle] = false;
926 }
927
928 MTCL_MQTT_PRINT(100, "connected to: %s\n", (manager_id + MQTT_CONNECTION_TOPIC).c_str());
929 return handle;
930 }

941 void notify_yield(Handle* h) override {
942 HandleMQTT* handle = reinterpret_cast<HandleMQTT*>(h);
943 bool enqueue_now = false;
944 {
945 REMOVE_CODE_IF(std::unique_lock l(shm));
946 auto it = connections.find(handle);
947 if(it == connections.end()) {
948 MTCL_MQTT_ERROR("Couldn't yield handle\n");
949 return;
950 }
951 // the handle is already IO-managed, skip
952 if (it->second) return;
953
954 // already something in the message buffer?
955 if (!handle->messages.empty()) {
956 enqueue_now = true;
957 } else {
958 // Check if handle still has some data to receive.
959 size_t size;
960 errno = 0;
961 const ssize_t r = handle->probe(size, false);
962 if (r >= 0) {
963 enqueue_now = true;
964 } else if (errno != EWOULDBLOCK && errno != EAGAIN) {
965 // propagate the error to the user level by re-queuing the handle
966 enqueue_now = true;
967 }
968 }
969 // If nothing is ready now, delegate polling to the IO thread.
970 if (!enqueue_now) it->second = true;
971 }
972 if (enqueue_now) addinQ(false, handle);
973 }
