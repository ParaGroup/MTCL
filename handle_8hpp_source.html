293void ConnType::setAsClosed(Handle* h, bool blockflag){
294 // Send EOS if not already sent
295 if(!h->closed_wr) {
296 h->close(true, false);
297 }
298
299 if (blockflag) {
300 // If EOS is still not received we wait for it, discarding pending messages
301 if(!h->closed_rd) {
302 size_t sz = 1;
303 while(true) {
304 if (h->probed.first) {
305 sz = h->probed.second;
306 } else {
307 if(h->probe(sz) == -1) {
308 MTCL_PRINT(100, "[MTCL]:", "ConnType::setAsClosed probe error\n");
309 return;
310 }
311 }
312 if(sz == 0) break;
313 MTCL_PRINT(100, "[MTCL]:", "Spurious message received of size %ld on handle with name %s!\n", sz, h->getName().c_str());
314 std::vector<char> buff(sz);
315 if(h->receive(buff.data(), sz) == -1) {
316 MTCL_PRINT(100, "[MTCL]:", "ConnType::setAsClosed receive error\n");
317 return;
318 }
319 }
320 }
321 }
322
323 // Finally closing the handle
324 h->close(false, true);
